function CONV_INTEGER(ARG: STD_LOGIC_VECTOR) return INTEGER is
  variable result: INTEGER;
begin
  assert ARG'length <= 32
    report "ARG is too large in CONV_INTEGER"
    severity FAILURE;
  result := 0;
  for i in ARG'range loop
    if i /= ARG'left then
      result := result * 2;
      if tbl_BINARY(ARG(i)) = '1' then
        result := result + 1;
      end if;
    end if;
  end loop;
  return result;
end;
